3    public interface Command {
4    	/**
5    	 * 这个方法是一个返回结果为空的方法。 实际项目中，可以根据需求设计多个不同的方法
6    	 */
7    	void execute();
8    }
9    
10   class ConcreteCommand implements Command {
11   
12   	private Receiver receiver; // 命令的真正的执行者
13   
14   	public ConcreteCommand(Receiver receiver) {
15   		super();
16   		this.receiver = receiver;
17   	}
18   
19   	@Override
20   	public void execute() {
21   		// 命令真正执行前或后，执行相关的处理！
22   		receiver.action();
23   	}
24   
25   }
