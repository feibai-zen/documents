3    import java.lang.reflect.Proxy;
4    
5    /**
6     * 动态代理与静态代理相比较，最大的好处是接口中声明的所有方法都被转移到调用处理器一个集中的方法中处理（InvocationHandler.invoke）。这样，在
7     * 接口方法数量比较多的时候，就可以进行灵活处理，而不需要像静态代理那样每一个方法进行中转。而且动态代理的应用使我们的类职责更加单一，复用性更强
8     */
9    public class ProxyDynamicClient {
10     public static void main(String[] args) {
11   
12       Star realStar = new Jay();
13       StarHandler handler = new StarHandler(realStar);
14   
15       Star proxy = (Star) Proxy.newProxyInstance(ClassLoader.getSystemClassLoader(), new Class[]{Star.class}, handler);
16   
17       proxy.bookTicket();
18       System.out.println("=============================");
19       proxy.sing();
20     }
21   
22     public void test() {
23       Star star = new Jay();
24       StarHandler handler = new StarHandler(star);
25   
26       Star proxy = (Star) Proxy.newProxyInstance(ClassLoader.getSystemClassLoader(), new Class[]{Star.class}, handler);
27       proxy.bookTicket();

29   
30     }
