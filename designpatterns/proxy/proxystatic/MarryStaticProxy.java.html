3    /**
4     * 使用的场景，一项任务，需要交给谁去做。为我办事，我是主体，代理负责周边辅助性工作。
5     * 比如结婚：结婚的人是我，我可以找婚庆公司做代理。我和婚庆公司怎么联系起来呢？靠的是
6     * 结婚这个事情，所以接口是结婚这个动作。代理要代理什么动作，就继承什么接口。
7     * 
8     * 抽象的：结婚的人是一个群体，视为核心对象，做的动作是核心动作。 代理处理的是通用的事
9     * 情，代理角色能力强大之后，也可能反客为主，成为重要的角色。
10    *
11    *
12    * 核心角色注入的方式1）构造方法  2）spring依赖注入
13    * 
14    * @author feibai
15    *
16    */
17   public class MarryStaticProxy {
18   
19   	public static void main(String[] args) {
20   		MarryCompany company = new MarryCompany(new You());
21   		company.marry();
22   	}
23   }
24   
25   interface Marry {
26   	void marry();
27   }
28   
29   class You extends Person {
30   	@Override
31   	public void marry() {
32   		System.out.println("I am marring...");
33   	}
34   
35   }
36   
37   class Person implements Marry {
38   	@Override
39   	public void marry() {
40   	}
41   }
42   
43   class MarryCompany implements Marry {
44   	private Person marryTarget;
45   
46   	public MarryCompany(Person marryTarget) {
47   		this.marryTarget = marryTarget;
48   	}
49   
50   	@Override
51   	public void marry() {
52   		prepareMarry();
53   		marryTarget.marry();
54   		endingMarry();
55   	}
56   
57   	private void prepareMarry() {
58   		System.out.println("婚礼前准备....");
59   	}
60   
61   	private void endingMarry() {
62   		System.out.println("婚礼后准备....");
63   	}
64   
65   }
