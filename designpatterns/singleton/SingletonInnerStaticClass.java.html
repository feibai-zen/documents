1    package com.feibai.study.demos.designpatterns.singleton;
2    
3    /**
4     * 静态内部类实现单例模式
5     * <p>
6     * 1.外部类没有static属性，则不会像饿汉式那样立即加载对象
7     * 2.只有调用getInstance()，才会加载静态内部类。加载类时是线程安全的。instance是static final类
8     * 型，保证了内存中只有一个实例存在，而且只能被赋值一次，从而保证了线程安全性。
9     * 3.兼备了高效调用和延迟加载的优势。加载SingletonInnerStaticClass类时，不会初始化静态内部类，因此具有懒加载。
10    *
11    * @author feibai
12    */
13   public class SingletonInnerStaticClass {
14   
15     private SingletonInnerStaticClass() {
16     }
17   
18     private static class Inner {
19       //静态内部类可以访问外层类
20       private static final SingletonInnerStaticClass INSTANCE =
21           new SingletonInnerStaticClass();
22     }
23   
24     public static SingletonInnerStaticClass getInstance() {
25   
26       return Inner.INSTANCE;
27     }
28   
29   }
