3    /**
4     * 懒汉式 --1）延迟加载，资源利用率高了；2） 缺点：每次调用getInstance()方法都要同步，并发效率低
5     *
6     * @author feibai
7     */
8    public class SingletonLazy {
9      private static SingletonLazy instance;
10   
11     // 私有构造器
12     private SingletonLazy() {
13   
14     }
15   
16     // 静态获取方法
17     public static synchronized SingletonLazy getInstance() {
18       if (null == instance) {
19         instance = new SingletonLazy();
20       }
21       return instance;
22     }
23   
24     public static void main(String[] args) {
25       System.out.println(SingletonLazy.getInstance());
26       System.out.println(SingletonLazy.getInstance());
27       System.out.println(SingletonLazy.getInstance());
28     }
29   
30   }
